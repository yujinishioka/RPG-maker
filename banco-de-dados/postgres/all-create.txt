/* create */

create table if not exists alinhamento(
	id_alinhamento int generated always as identity ( start with 1 increment by 1 ) primary key,
	tipo varchar(50) not null,
	explicacao varchar(5000) not null
);

create table if not exists personagem(
	id_personagem int generated always as identity ( start with 1 increment by 1 ) primary key,
	nome_jogador varchar(100) not null,
	nome_personagem varchar(100) not null,
	titulo varchar(200),
	nivel numeric(4) not null,
	experiencia numeric(10, 2) not null,
	iniciativa numeric(4) not null,
	deslocamento numeric(8) not null,
	armadura numeric(4) not null,
	vida numeric(8) not null,
	proficiencia numeric(4) not null,
	inspiracao boolean,
	idade varchar(50),
	altura varchar(50),
	peso varchar(50),
	cor_olhos varchar(50),
	cor_pele varchar(50),
	cor_cabelo varchar(50),
	personalidade varchar(2000),
	ideais varchar(2000),
	vinculos varchar(2000),
	fraquezas varchar(2000),
	caracteristicas varchar(2000),
	outras_proficiencias varchar(2000),
	aliados varchar(2000),
	historia_personagem varchar(2000),
	id_alinhamento int not null,
	foreign key(id_alinhamento)
		references alinhamento(id_alinhamento)
);

create table if not exists status(
	id_status int generated always as identity ( start with 1 increment by 1 ) primary key,
	ability varchar (50) not null,
	valor numeric (4) not null,
	id_personagem int not null,
	foreign key(id_personagem)
		references personagem(id_personagem)
);
